### Test

* Write `tools/validate.py` to run jsonschema validation over all YAML.
* Run `python tools/validate.py` → all configs valid.

Example implementation (complete but minimal):

```python
#!/usr/bin/env python3
import json
import sys
from pathlib import Path

import yaml
from jsonschema import validate, ValidationError


REPO_ROOT = Path(__file__).resolve().parents[1]


def load_schema(name: str) -> dict:
    with open(REPO_ROOT / "schemas" / name, "r", encoding="utf-8") as f:
        return json.load(f)


def load_yaml(path: Path) -> dict:
    with open(path, "r", encoding="utf-8") as f:
        return yaml.safe_load(f)


def validate_hosts() -> list[str]:
    schema = load_schema("host-schema.json")
    errors = []
    for path in (REPO_ROOT / "config" / "hosts").rglob("host.yaml"):
        data = load_yaml(path)
        try:
            validate(data, schema)
        except ValidationError as e:
            errors.append(f"Host config {path}: {e.message}")
    return errors


def validate_services() -> list[str]:
    schema = load_schema("service-schema.json")
    errors = []
    for path in (REPO_ROOT / "config" / "services").rglob("service.yaml"):
        data = load_yaml(path)
        try:
            validate(data, schema)
        except ValidationError as e:
            errors.append(f"Service config {path}: {e.message}")
    return errors


def validate_mapping() -> list[str]:
    schema = load_schema("mapping-schema.json")
    path = REPO_ROOT / "config" / "mapping" / "hosts-services.yaml"
    data = load_yaml(path)
    try:
        validate(data, schema)
    except ValidationError as e:
        return [f"Mapping {path}: {e.message}"]
    return []


def main() -> int:
    all_errors: list[str] = []
    all_errors += validate_hosts()
    all_errors += validate_services()
    all_errors += validate_mapping()

    if all_errors:
        print("Validation failed:")
        for e in all_errors:
            print(f"- {e}")
        return 1

    print("All configs valid ✔️")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

## Phase 2 – Templating & rendering (envsubst-style, declarative)

### 2.1 – Simple envsubst helper

We’ll use `${VAR}` placeholders in templates and substitute from a dict built from host + service + global config.

`tools/hashing.py`:

```python
#!/usr/bin/env python3
import hashlib
from pathlib import Path


STATE_DIR = Path("/var/lib/abhaile/state")


def hash_bytes(data: bytes) -> str:
    return hashlib.sha256(data).hexdigest()


def hash_file(path: Path) -> str:
    return hash_bytes(path.read_bytes())


def state_path(kind: str, key: str) -> Path:
    safe = key.replace("/", "_")
    return STATE_DIR / kind / f"{safe}.sha256"


def has_changed(kind: str, key: str, new_hash: str) -> bool:
    p = state_path(kind, key)
    if not p.exists():
        return True
    old = p.read_text(encoding="utf-8").strip()
    return old != new_hash


def record_hash(kind: str, key: str, new_hash: str) -> None:
    p = state_path(kind, key)
    p.parent.mkdir(parents=True, exist_ok=True)
    p.write_text(new_hash + "\n", encoding="utf-8")
```

`tools/orchestrate.py` (core rendering logic):

```python
#!/usr/bin/env python3
import argparse
import os
from pathlib import Path
from string import Template
from typing import Dict, Any, List

import yaml

from hashing import hash_bytes, has_changed, record_hash


REPO_ROOT = Path(__file__).resolve().parents[1]
RENDER_DIR = Path("/var/lib/abhaile/rendered")


def load_yaml(path: Path) -> Any:
    with open(path, "r", encoding="utf-8") as f:
        return yaml.safe_load(f)


def build_context(host_id: str) -> Dict[str, Any]:
    host_cfg = load_yaml(REPO_ROOT / "config" / "hosts" / host_id / "host.yaml")
    mapping = load_yaml(REPO_ROOT / "config" / "mapping" / "hosts-services.yaml")
    services = {}
    for svc_name in mapping["hosts"][host_id]["services"]:
        svc_path = REPO_ROOT / "config" / "services" / svc_name / "service.yaml"
        services[svc_name] = load_yaml(svc_path)

    ctx: Dict[str, Any] = {}
    ctx["HOST_ID"] = host_cfg["id"]
    ctx["HOSTNAME"] = host_cfg["hostname"]
    ctx["ROLES"] = ",".join(host_cfg["roles"])
    ctx["PRIMARY_IP"] = host_cfg["vlans"]["primary"]["address"].split("/")[0]
    ctx["DNS_SERVERS"] = ",".join(host_cfg["dns_servers"])
    ctx["NTP_SERVERS"] = ",".join(host_cfg["ntp_servers"])
    # You can also flatten services into variables if needed.
    ctx["SERVICES"] = services
    return ctx


def envsubst_string(template_str: str, context: Dict[str, Any]) -> str:
    # Only substitute simple string keys from context
    simple_ctx = {k: v for k, v in context.items() if isinstance(v, str)}
    return Template(template_str).safe_substitute(simple_ctx)


def render_template(src: Path, dst: Path, context: Dict[str, Any]) -> bool:
    raw = src.read_text(encoding="utf-8")
    rendered = envsubst_string(raw, context)
    data = rendered.encode("utf-8")
    h = hash_bytes(data)
    key = str(dst.relative_to("/"))
    if not has_changed("files", key, h):
        return False
    dst.parent.mkdir(parents=True, exist_ok=True)
    dst.write_bytes(data)
    record_hash("files", key, h)
    return True


def render_host(host_id: str) -> List[Path]:
    ctx = build_context(host_id)
    changed: List[Path] = []

    # systemd-networkd
    tmpl_dir = REPO_ROOT / "config" / "hosts" / "common" / "systemd-networkd"
    for tmpl in tmpl_dir.glob("*.tmpl"):
        dst_name = tmpl.name.replace(".tmpl", "")
        dst = RENDER_DIR / host_id / "etc/systemd/network" / dst_name
        if render_template(tmpl, dst, ctx):
            changed.append(dst)

    # resolved
    res_tmpl = (
        REPO_ROOT
        / "config"
        / "hosts"
        / "common"
        / "systemd-resolved"
        / "resolved.conf.tmpl"
    )
    res_dst = RENDER_DIR / host_id / "etc/systemd/resolved.conf"
    if render_template(res_tmpl, res_dst, ctx):
        changed.append(res_dst)

    # you can extend for user configs, etc.
    return changed


def render_services(host_id: str) -> List[Path]:
    ctx = build_context(host_id)
    mapping = load_yaml(REPO_ROOT / "config" / "mapping" / "hosts-services.yaml")
    changed: List[Path] = []
    for svc_name in mapping["hosts"][host_id]["services"]:
        svc_dir = REPO_ROOT / "config" / "services" / svc_name
        # env
        env_dir = svc_dir / "env"
        for tmpl in env_dir.glob("*.tmpl"):
            dst_name = tmpl.name.replace(".tmpl", "")
            dst = RENDER_DIR / host_id / "opt/abhaile/services" / svc_name / "env" / dst_name
            if render_template(tmpl, dst, ctx):
                changed.append(dst)
        # config
        cfg_dir = svc_dir / "config"
        for tmpl in cfg_dir.glob("*.tmpl"):
            dst_name = tmpl.name.replace(".tmpl", "")
            dst = (
                RENDER_DIR
                / host_id
                / "opt/abhaile/services"
                / svc_name
                / "config"
                / dst_name
            )
            if render_template(tmpl, dst, ctx):
                changed.append(dst)
    return changed


def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("phase", choices=["render", "host", "services", "all"])
    parser.add_argument("--host", required=True)
    parser.add_argument("--apply", action="store_true")
    args = parser.parse_args()

    host_id = args.host

    if args.phase in ("render", "all"):
        changed_host = render_host(host_id)
        changed_services = render_services(host_id)
        print(f"Rendered host files: {len(changed_host)}")
        print(f"Rendered service files: {len(changed_services)}")

    if args.apply:
        # Defer to bash apply scripts in later phases
        os.system(f"tools/bash/host_apply.sh {host_id}")
        os.system(f"tools/bash/quadlet_apply.sh {host_id}")
        os.system(f"tools/bash/caddy_apply.sh {host_id}")
        os.system(f"tools/bash/coredns_apply.sh {host_id}")

    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

`tools/bash/host_apply.sh` (minimal but functional):

```bash
#!/usr/bin/env bash
set -euo pipefail
source "$(dirname "$0")/lib.sh"

host="${1:?host id required}"

host_root="${render_root}/${host}"

apply_file() {
  local src="$1"
  local dst="$2"
  local unit="$3"   # systemd unit to restart/reload or empty
  local reload_cmd="${4:-}" # override reload command

  if [[ ! -f "$src" ]]; then
    return 0
  fi

  mkdir -p "$(dirname "$dst")"
  local changed=0

  if [[ ! -f "$dst" ]] || ! cmp -s "$src" "$dst"; then
    changed=1
    log "Updating $dst"
    install -m 0644 "$src" "$dst"
  fi

  if [[ $changed -eq 1 && -n "$unit" ]]; then
    if [[ -n "$reload_cmd" ]]; then
      log "Reload command for $unit: $reload_cmd"
      eval "$reload_cmd"
    else
      log "Restarting $unit"
      systemctl restart "$unit"
    fi
  fi
}

main() {
  # systemd-networkd
  for f in "${host_root}/etc/systemd/network/"*; do
    [[ -e "$f" ]] || continue
    dst="/etc/systemd/network/$(basename "$f")"
    apply_file "$f" "$dst" "systemd-networkd.service" "systemctl restart systemd-networkd.service"
  done

  # resolved
  if [[ -f "${host_root}/etc/systemd/resolved.conf" ]]; then
    apply_file "${host_root}/etc/systemd/resolved.conf" \
      "/etc/systemd/resolved.conf" \
      "systemd-resolved.service" \
      "systemctl restart systemd-resolved.service"
  fi

  # you can extend here for users, sudoers, nftables, etc.
}

main "$@"
```

### 3.2 – Quadlets, auto-prefixing, and selective restarts

You want:

* copy from `services/<name>/quadlets/*.container/.image/.volume/.build` (rendered path)
* into `/etc/containers/systemd/<service>-<srcname>`
* restart only changed units
* `daemon-reload` once

`tools/bash/quadlet_apply.sh`:

```bash
#!/usr/bin/env bash
set -euo pipefail
source "$(dirname "$0")/lib.sh"

host="${1:?host id required}"
host_root="${render_root}/${host}"

quadlet_src_root="${host_root}/opt/abhaile/services"
quadlet_dst_root="/etc/containers/systemd"

main() {
  local changed_units=()
  local daemon_reload=0

  while IFS= read -r -d '' f; do
    # f looks like .../services/<svc>/quadlets/<file>.container
    local svc
    svc="$(echo "$f" | awk -F'/services/' '{print $2}' | cut -d'/' -f1)"
    local basename
    basename="$(basename "$f")"
    local dst="${quadlet_dst_root}/${svc}-${basename}"

    mkdir -p "$quadlet_dst_root"

    local changed=0
    if [[ ! -f "$dst" ]] || ! cmp -s "$f" "$dst"; then
      log "Updating quadlet $dst"
      install -m 0644 "$f" "$dst"
      changed=1
      daemon_reload=1
    fi

    if [[ $changed -eq 1 ]]; then
      local unit="${svc}-${basename}"
      # strip .container/.volume/.image/.build for systemd unit
      unit="${unit%.*}"
      changed_units+=("${unit}.service")
    fi
  done < <(find "${quadlet_src_root}" -type f -path "*/quadlets/*" -print0)

  if [[ $daemon_reload -eq 1 ]]; then
    log "Running systemctl daemon-reload"
    systemctl daemon-reload
  fi

  # deduplicate units
  mapfile -t uniq_units < <(printf '%s\n' "${changed_units[@]}" | sort -u)
  for u in "${uniq_units[@]}"; do
    # containers: prefer restart; networks/volumes: no restart
    if [[ "$u" == *.service ]]; then
      log "Restarting container unit $u"
      systemctl restart "$u"
    fi
  done
}

main "$@"
```

## Phase 4 – Dynamic Caddy & CoreDNS

### 4.1 – Caddyfile build (from services + mapping)

`tools/caddy_build.py` reads all `service.yaml` and builds **one** internal Caddyfile and one DMZ Caddyfile:

Concept:

* For every service with `fqdn_internal`, create site block:

  * reverse_proxy to its `/32` IP
  * apply Authelia middleware based on service metadata (you can add `auth: required|optional|none` to service config)
* For DMZ, only for those with `fqdn_public`.

Template snippet (Caddy 2):

```caddy
{$FQDN_INTERNAL} {
    encode gzip
    reverse_proxy {$UPSTREAM}
    tls internal
    @authed {
        not path /authelia
    }
    forward_auth @authed {
        uri https://authelia.abhaile.home.arpa/api/verify?rd=https://{$FQDN_INTERNAL}
        copy_headers Remote-User Remote-Groups
    }
}
```

Your `caddy_build.py` would produce `RENDER_DIR/<host>/etc/caddy/Caddyfile` and then `caddy_apply.sh`:

```bash
#!/usr/bin/env bash
set -euo pipefail
source "$(dirname "$0")/lib.sh"

host="${1:?host id required}"
src="/var/lib/abhaile/rendered/${host}/etc/caddy/Caddyfile"
dst="/etc/caddy/Caddyfile"

if [[ ! -f "$src" ]]; then
  exit 0
fi

changed=0
if [[ ! -f "$dst" ]] || ! cmp -s "$src" "$dst"; then
  log "Updating Caddyfile"
  install -m 0644 "$src" "$dst"
  changed=1
fi

if [[ $changed -eq 1 ]]; then
  log "Reloading caddy"
  caddy validate --config "$dst"
  systemctl reload caddy.service || systemctl restart caddy.service
fi
```

## Phase 7 – INVENTORY.md generator (services, hosts, ports, FQDNs)

### 7.1 – Inventory generator

`tools/inventory_gen.py`:

```python
#!/usr/bin/env python3
import subprocess
from pathlib import Path

import yaml

REPO_ROOT = Path(__file__).resolve().parents[1]


def get_git_rev() -> str:
    try:
        return (
            subprocess.check_output(
                ["git", "rev-parse", "--short", "HEAD"], cwd=REPO_ROOT
            )
            .decode()
            .strip()
        )
    except Exception:
        return "unknown"


def load_yaml(path: Path):
    with open(path, "r", encoding="utf-8") as f:
        return yaml.safe_load(f)


def build_inventory() -> str:
    mapping = load_yaml(REPO_ROOT / "config" / "mapping" / "hosts-services.yaml")

    header = (
        "# INVENTORY\n\n"
        f"Generated from config at commit `{get_git_rev()}`.\n\n"
        "| Service | Host | VLAN | IP | Ports | Internal FQDN | Public FQDN | Notes |\n"
        "|--------|------|------|----|-------|----------------|-------------|-------|\n"
    )

    rows = []
    for host, info in mapping["hosts"].items():
        for svc_name in info["services"]:
            svc_cfg = load_yaml(REPO_ROOT / "config" / "services" / svc_name / "service.yaml")
            ip = svc_cfg.get("bind_ip", "") or ""
            vlan = svc_cfg.get("vlan", "")
            ports = ", ".join(svc_cfg.get("ports", []))
            fqdn_int = svc_cfg.get("fqdn_internal") or ""
            fqdn_pub = svc_cfg.get("fqdn_public") or ""
            notes = svc_cfg.get("notes", "")

            rows.append(
                f"| {svc_name} | {host} | {vlan} | {ip} | {ports} | {fqdn_int} | {fqdn_pub} | {notes} |"
            )

    return header + "\n".join(sorted(rows)) + "\n"


def main() -> int:
    inventory_md = build_inventory()
    (REPO_ROOT / "INVENTORY.md").write_text(inventory_md, encoding="utf-8")
    print("INVENTORY.md updated.")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

```python
#!/usr/bin/env python3
import argparse
from pathlib import Path

REPO_ROOT = Path(__file__).resolve().parents[1]


TPL = """#!/usr/bin/env bash
set -euo pipefail

HOST="{host}"
REPO_URL="{repo_url}"

apt-get update
apt-get install -y git python3 python3-pip jq curl sops podman

mkdir -p /opt
if [[ ! -d /opt/abhaile ]]; then
  git clone "$REPO_URL" /opt/abhaile
fi

cp /opt/abhaile/systemd/git-sync@.service /etc/systemd/system/
cp /opt/abhaile/systemd/git-sync@.timer /etc/systemd/system/

systemctl daemon-reload
systemctl enable --now git-sync@${{HOST}}.timer

# initial render + apply
cd /opt/abhaile
python3 tools/orchestrate.py render --host "$HOST" --apply
"""

def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--host", required=True)
    ap.add_argument("--repo-url", required=False, default="git@github.com:you/abhaile.git")
    ap.add_argument("--output", required=True)
    args = ap.parse_args()

    content = TPL.format(host=args.host, repo_url=args.repo_url)
    Path(args.output).parent.mkdir(parents=True, exist_ok=True)
    Path(args.output).write_text(content, encoding="utf-8")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```
